#### 步骤1: 用户授权检查

```python
async def _handle_message(self, event: MessageEvent) -> Optional[str]:
    source = event.source

    # 内部事件跳过授权
    if getattr(event, "internal", False):
        pass
    elif source.user_id is None:
        return None  # 无用户身份的消息静默丢弃
    elif not self._is_user_authorized(source):
        # DM: 发送配对码; 群组: 静默忽略
        if source.chat_type == "dm":
            code = self.pairing_store.generate_code(platform, user_id, user_name)
            if code:
                await adapter.send(chat_id, f"Your pairing code: `{code}`")
        return None
```

**配对系统安全特性**:
- 8位配对码，32字符无歧义字母表 (排除0/O/1/I)
- `secrets.choice()` 加密随机性
- 1小时码过期
- 每平台最多3个待处理码
- 每用户每10分钟1次请求速率限制
- 5次失败审批尝试后锁定1小时
- 文件权限 `chmod 0600`

```python
async def _run_agent(self, message, context_prompt, history, source, session_id, ...):
    # 代理模式检查
    if self._get_proxy_url():
        return await self._run_agent_via_proxy(...)

    # 解析 runtime credentials
    model, runtime_kwargs = self._resolve_session_agent_runtime(source=source)

    # 创建或复用缓存的 AIAgent
    cache_key = session_key
    cached = self._agent_cache.get(cache_key)
    if cached and _agent_config_matches(cached, model, runtime_kwargs):
        agent = cached[0]  # 复用缓存Agent (保留prompt-cache)
    else:
        agent = AIAgent(
            **runtime_kwargs,
            model=model,
            max_iterations=max_iterations,
            session_id=session_id,
            stream_delta_callback=consumer.on_delta if consumer else None,
        )
        self._agent_cache[cache_key] = (agent, config_signature)

    # 在线程池中运行 (不阻塞事件循环)
    loop = asyncio.get_event_loop()
    result = await loop.run_in_executor(
        None,
        lambda: agent.run_conversation(
            user_message=message,
            conversation_history=history,
            system_prompt=context_prompt,
        ),
    )
```

### 2.3 Agent缓存

```python
# Agent缓存: 保留prompt-cache，避免每次消息重建系统提示
# Key: session_key, Value: (AIAgent, config_signature_str)
self._agent_cache: Dict[str, tuple] = {}
self._agent_cache_lock = threading.Lock()

# 缓存命中条件:
def _agent_config_matches(cached, agent_model, runtime_kwargs) -> bool:
    cached_agent, cached_sig = cached
    # 1. 模型名称匹配
    # 2. 提供商凭证匹配
    # 3. 会话模型覆盖匹配
    return cached_agent.model == agent_model and ...
```

**Prompt-Cache 优化原理**:
- Anthropic等提供商支持prompt caching: 相同前缀的请求可复用缓存
- 每次创建新AIAgent会重建系统提示(包含memory)，破坏前缀缓存
- 缓存Agent后，后续消息的系统提示部分可命中缓存，节省约10x成本
- `/new`, `/model` 等命令会清除缓存 (`_evict_cached_agent`)

## 5. 流式消费 (GatewayStreamConsumer)

### 5.1 架构设计

```
Agent线程 (同步)          事件循环 (异步)
    |                         |
    v                         v
on_delta(text)  ──→  queue.Queue  ──→  run() 任务
    |                         |
    |                    +----+----+
    |                    |         |
    |                _send_or_edit()
    |                    |
    |              adapter.edit_message()
    |                    |
    v                    v
  完成  ──→  finish()  ──→  最终编辑
```

```python
class GatewayStreamConsumer:
    """同步回调 → 异步队列桥接"""

    _DONE = object()           # 流完成信号
    _NEW_SEGMENT = object()    # 工具边界信号
    _COMMENTARY = object()     # 中间助手消息

    def __init__(self, adapter, chat_id, config, metadata=None):
        self._queue: queue.Queue = queue.Queue()
        self._accumulated = ""
        self._message_id: Optional[str] = None
        self._edit_supported = True
        self._current_edit_interval = self.cfg.edit_interval  # 自适应退避

    def on_delta(self, text: str) -> None:
        """线程安全回调 — 从Agent工作线程调用"""
        if text:
            self._queue.put(text)
        elif text is None:
            self.on_segment_break()  # 工具边界

    def finish(self) -> None:
        """信号流完成"""
        self._queue.put(_DONE)
```

**编辑传输模式**:
1. 先发送初始消息 (`adapter.send`)
2. 获取 `message_id`
3. 逐步 `adapter.edit_message` 更新内容
4. 添加光标 `▉` 表示正在输入
5. 完成时移除光标

```python
    _OPEN_THINK_TAGS = (
        "<REASONING_SCRATCHPAD>", "<think>", "<reasoning>",
        "<THINKING>", "<thinking>", "<thought>",
    )
    _CLOSE_THINK_TAGS = (
        "</REASONING_SCRATCHPAD>", "</think>", "</reasoning>",
        "</THINKING>", "</thinking>", "</thought>",
    )

    def _filter_and_accumulate(self, text: str) -> None:
        """状态机处理，过滤推理/思考块"""
        buf = self._think_buffer + text
        self._think_buffer = ""

        while buf:
            if self._in_think_block:
                # 查找最早的关闭标签
                for tag in self._CLOSE_THINK_TAGS:
                    idx = buf.find(tag)
                    if idx != -1:
                        self._in_think_block = False
                        buf = buf[idx + len(tag):]
                        break
                else:
                    # 未找到关闭标签 — 保留可能的部分标签前缀
                    max_tag = max(len(t) for t in self._CLOSE_THINK_TAGS)
                    self._think_buffer = buf[-max_tag:] if len(buf) > max_tag else buf
                    return
            else:
                # 查找最早的开始标签 (仅在块边界)
                for tag in self._OPEN_THINK_TAGS:
                    idx = buf.find(tag)
                    if idx != -1:
                        is_boundary = (
                            idx == 0 and (not self._accumulated or self._accumulated.endswith("\n"))
                        )
                        if is_boundary:
                            self._accumulated += buf[:idx]
                            self._in_think_block = True
                            buf = buf[idx + len(tag):]
                            break
                else:
                    # 检查尾部的部分标签
                    held_back = 0
                    for tag in self._OPEN_THINK_TAGS:
                        for i in range(1, len(tag)):
                            if buf.endswith(tag[:i]) and i > held_back:
                                held_back = i
                    if held_back:
                        self._accumulated += buf[:-held_back]
                        self._think_buffer = buf[-held_back:]
                    else:
                        self._accumulated += buf
                    return
```

```python
class HookRegistry:
    def discover_and_load(self) -> None:
        # 1. 注册内置钩子 (boot_md)
        self._register_builtin_hooks()

        # 2. 扫描 hooks 目录
        for hook_dir in sorted(HOOKS_DIR.iterdir()):
            manifest_path = hook_dir / "HOOK.yaml"
            handler_path = hook_dir / "handler.py"

            # 3. 加载清单
            manifest = yaml.safe_load(manifest_path.read_text())
            events = manifest.get("events", [])

            # 4. 动态加载处理器模块
            spec = importlib.util.spec_from_file_location(
                f"hermes_hook_{hook_name}", handler_path
            )
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)

            # 5. 注册处理器
            handle_fn = getattr(module, "handle", None)
            for event in events:
                self._handlers.setdefault(event, []).append(handle_fn)

    async def emit(self, event_type: str, context=None) -> None:
        # 精确匹配 + 通配符匹配
        handlers = list(self._handlers.get(event_type, []))

        # "command:*" 匹配 "command:reset"
        if ":" in event_type:
            base = event_type.split(":")[0]
            wildcard_key = f"{base}:*"
            handlers.extend(self._handlers.get(wildcard_key, []))

        for fn in handlers:
            try:
                result = fn(event_type, context)
                if asyncio.iscoroutine(result):
                    await result
            except Exception as e:
                print(f"[hooks] Error in handler for '{event_type}': {e}")
```

```python
        try:
            await self._read_events()
            backoff_idx = 0
        except Exception:
```

```python
APP_CMD_UPLOAD_MEDIA_INIT = "aibot_upload_media_init"
APP_CMD_UPLOAD_MEDIA_CHUNK = "aibot_upload_media_chunk"
APP_CMD_UPLOAD_MEDIA_FINISH = "aibot_upload_media_finish"

UPLOAD_CHUNK_SIZE = 512 * 1024  # 512KB
MAX_UPLOAD_CHUNKS = 100

async def _upload_large_file(self, file_path, file_type):
    """分块上传大文件"""
    file_size = os.path.getsize(file_path)
    total_chunks = (file_size + UPLOAD_CHUNK_SIZE - 1) // UPLOAD_CHUNK_SIZE

    # 1. 初始化上传
    init_resp = await self._send_request("aibot_upload_media_init", {
        "file_type": file_type,
        "file_size": file_size,
        "total_chunks": total_chunks,
    })
    upload_id = init_resp["body"]["upload_id"]

    # 2. 分块上传
    with open(file_path, "rb") as f:
        for chunk_idx in range(total_chunks):
            chunk_data = f.read(UPLOAD_CHUNK_SIZE)
            await self._send_request("aibot_upload_media_chunk", {
                "upload_id": upload_id,
                "chunk_index": chunk_idx,
                "data": base64.b64encode(chunk_data).decode(),
            })

    # 3. 完成上传
    finish_resp = await self._send_request("aibot_upload_media_finish", {
        "upload_id": upload_id,
    })
    return finish_resp["body"]["media_key"]
```

### 10.5 遇到的问题和解决方案

**问题1: 并发消息竞态条件**
- **现象**: 两个消息同时到达，都通过了"已在运行"守卫，导致重复Agent
- **解决**: `_AGENT_PENDING_SENTINEL` 哨兵机制，在await点之前立即放置哨兵

**问题2: 会话卡死循环 (#7536)**
- **现象**: 网关重启后，卡死的会话被盲目恢复，导致无限循环
- **解决**: 启动时挂起最近活跃的会话，`/stop`命令挂起会话，自动重置

**问题3: 上下文溢出**
- **现象**: 长会话超过模型上下文窗口，API返回400错误
- **解决**: 会话卫生自动压缩 (85%阈值)，硬安全阀 (400条消息)，自动重置

**问题4: 流式编辑洪泛控制**
- **现象**: Telegram等平台限制消息编辑频率，导致编辑失败
- **解决**: 自适应退避 (间隔翻倍)，回退到发送新消息，Think块过滤减少编辑次数

**问题5: 飞书客户端消息分片**
- **现象**: 飞书客户端将长消息分片发送，导致Agent收到不完整消息
- **解决**: 文本批量聚合 (0.6s延迟)，分片检测 (3900字符阈值)，延长等待 (2.0s)

**问题6: ContextVar替代os.environ**
- **现象**: 并发消息处理时，os.environ的会话变量被覆盖
- **解决**: 使用Python ContextVar实现任务级隔离，每个asyncio任务独立副本

**问题7: PII脱敏**
- **现象**: 用户ID等敏感信息发送给LLM，存在隐私风险
- **解决**: 可选PII脱敏模式，SHA256哈希用户/聊天ID，保留平台前缀

**问题8: 崩溃恢复**
- **现象**: 网关崩溃后，后台进程状态丢失
- **解决**: 进程检查点恢复，会话挂起机制，干净关闭标记
